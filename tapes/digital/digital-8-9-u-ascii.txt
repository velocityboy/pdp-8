
/LOGICAL SHIFT RIGHT SUBROUTINES
/SINGLE AND DOUBLE PRECISION
/ENTER WITH -N IN AC
/DATA ADDRESS FOLLOWS CALLING JMS
/RETURN WITH DATA IN AC
/MOST SIGNIFICANT PART FOR DOUBLE
/LEAST SIG. PART FOR DOUBLE IN LESTSG

LSRSP,	0		/SINGLE PRECISION
	DCA TIMES
	TAD I LSRSP
	DCA COMMUN
	TAD I COMMUN
	CLL RAR		/SHIFT LOOP
	ISZ TIMES
	JMP .-2
	ISZ LSRSP		/EXIT
	JMP I LSRSP
LSRDP,	0		/DOUBLE PRECISION
	DCA TIMES
	TAD I LSRDP
	DCA COMMUN
	TAD I COMMUN
	DCA MOSTSG
	ISZ COMMUN
	TAD I COMMUN
	DCA LESTSG
SHIFT,	TAD MOSTSG	/SHIFT LOOP
	CLL RAR
	DCA MOSTSG
	TAD LESTSG
	RAR
	DCA LESTSG
	ISZ TIMES
	JMP SHIFT
	TAD MOSTSG	/EXIT
	ISZ LSRDP
	JMP I LSRDP
TIMES,	0
COMMUN,	0
MOSTSG,	0
LESTSG,	0
